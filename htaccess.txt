# LandingServizi / LavaggioDivani
# Redirect anti-404 per dominio potenzialmente già utilizzato
# File finale da copiare nella root del sito come: .htaccess
#
# Obiettivo:
# - mantenere funzionanti tutte le pagine reali generate dallo script;
# - reindirizzare qualsiasi vecchio URL inesistente verso la Home;
# - evitare che visitatori provenienti da vecchi link finiscano su una 404.
#
# Nota:
# Questo è un redirect generale. In futuro, se Google Search Console mostrerà vecchi URL rilevanti,
# sarà meglio creare redirect specifici verso la pagina più vicina, non sempre verso la Home.

RewriteEngine On

# Mantieni accessibili file e cartelle esistenti
RewriteCond %{REQUEST_FILENAME} -f [OR]
RewriteCond %{REQUEST_FILENAME} -d
RewriteRule ^ - [L]

# Reindirizza tutto ciò che non esiste alla Home
RewriteRule ^.*$ / [R=301,L]
